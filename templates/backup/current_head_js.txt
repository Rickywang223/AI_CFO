// ===== Search Function =====
function invSearchChange(val) {
    var dropdown = document.getElementById('invSearchDropdown');
    if (!val || val.length < 1) { dropdown.classList.remove('show'); return; }
    var matches = [];
    var lower = val.toLowerCase();
    // Search regions
    REGION_INVOICE_DATA.forEach(function(r) {
        if (r.name.indexOf(lower) >= 0 || r.id.indexOf(lower) >= 0) {
            matches.push({ name: r.name, type: '\u533a\u57df', detail: r.companyCount + '\u5bb6\u516c\u53f8' });
        }
    });
    // Search suppliers
    SUPPLIER_INVOICE_DATA.forEach(function(s) {
        if (s.name.indexOf(lower) >= 0) {
            matches.push({ name: s.name, type: '\u4f9b\u5e94\u5546', detail: '\u98ce\u9669\u7b49\u7ea7 ' + s.riskLevel + '\u7ea7' });
        }
    });
    // Search abnormal companies
    ABNORMAL_COMPANIES.highRisk.concat(ABNORMAL_COMPANIES.warning).forEach(function(c) {
        if (c.name.indexOf(lower) >= 0) {
            matches.push({ name: c.name, type: c.level === 'region' ? '\u533a\u57df' : '\u516c\u53f8', detail: '\u8ba4\u8bc1\u7387 ' + c.certRate + '%' });
        }
    });
    if (matches.length === 0) {
        dropdown.innerHTML = '<div style="padding:10px;font-size:11px;color:#bfbfbf;text-align:center;">\u65e0\u5339\u914d\u7ed3\u679c</div>';
        dropdown.classList.add('show');
        return;
    }
    var html = '';
    matches.slice(0, 8).forEach(function(m) {
        var highlighted = m.name.replace(new RegExp('(' + val + ')', 'gi'), '<span class="si-match">$1</span>');
        html += '<div class="inv-search-item" onclick=\'executeSuggestion("showRegionInvoiceDetail(\\"huazhong\\")")\'>' +
            '<span>' + highlighted + '</span> <span style="color:#8c8c8c;font-size:10px;">' + m.type + ' \u00b7 ' + m.detail + '</span>' +
        '</div>';
    });
    if (matches.length > 8) html += '<div style="padding:6px;font-size:10px;color:#bfbfbf;text-align:center;">\u8fd8\u6709 ' + (matches.length - 8) + ' \u4e2a\u7ed3\u679c...</div>';
    dropdown.innerHTML = html;
    dropdown.classList.add('show');
}
document.addEventListener('click', function(e) {
    if (!e.target.closest('.inv-search-wrapper')) {
        var d = document.getElementById('invSearchDropdown');
        if (d) d.classList.remove('show');
    }
});

function executeSuggestion(action) {
    if (action === 'batchCertifyHighRisk') {
        batchCertifyHighRisk();
    } else if (action.indexOf('showRegionInvoiceDetail') >= 0) {
        // Extract regionId
        var match = action.match(/"([^"]+)"/);
        if (match) showRegionInvoiceDetail(match[1]);
    } else {
        sendChat(action);
    }
}

function showFullCalendar() {
    var data = TAX_PERIOD;
    var year = 2026, month = 6;
    var firstDay = new Date(year, month, 1).getDay();
    var daysInMonth = new Date(year, month + 1, 0).getDate();
    var prevMonthDays = new Date(year, month, 0).getDate();
    var weekdays = ['\\u65e5','\\u4e00','\\u4e8c','\\u4e09','\\u56db','\\u4e94','\\u516d'];
    var now = new Date();
    var today = now.getDate();

    var events = {};
    data.taxes.forEach(function(t) {
        var day = parseInt(t.deadline.split('-')[2], 10);
        if (!events[day]) events[day] = [];
        events[day].push(t.type);
    });

    var calHtml = '';
    weekdays.forEach(function(w) { calHtml += '<div class="cal-weekday">' + w + '</div>'; });

    var startDay = firstDay === 0 ? 6 : firstDay - 1;
    for (var i = startDay - 1; i >= 0; i--) {
        calHtml += '<div class="cal-day other-month">' + (prevMonthDays - i) + '</div>';
    }

    for (var d = 1; d <= daysInMonth; d++) {
        var cls = 'cal-day';
        if (d === today) cls += ' today';
        if (events[d]) cls += ' event';
        calHtml += '<div class="' + cls + '">' + d;
        if (events[d]) {
            var dotType = events[d].length > 1 ? 'multi' : (events[d][0] === '\\u589e\\u503c\\u7a0e' ? 'vat' : (events[d][0] === '\\u4e2a\\u4eba\\u6240\\u5f97\\u7a0e' ? 'personal' : 'corp'));
            calHtml += '<div class="cd-dot ' + dotType + '"></div>';
        }
        calHtml += '</div>';
    }

    var modalOverlay = document.createElement('div');
    modalOverlay.className = 'cal-modal-overlay';
    modalOverlay.id = 'calModalOverlay';
    modalOverlay.onclick = function(e) { if (e.target === this) this.remove(); };

    modalOverlay.innerHTML =
        '<div class="cal-modal">' +
            '<div class="cal-modal-header">' +
                '<div class="cal-modal-title">\\ud83d\\udcc5 \\u5f81\\u671f\\u65e5\\u5386 &nbsp;2026\\u5e747\\u6708</div>' +
                '<button class="cal-modal-close" onclick="document.getElementById(\u0027calModalOverlay\u0027).remove()">\u2715</button>' +
            '</div>' +
            '<div class="cal-grid">' + calHtml + '</div>' +
            '<div class="cal-legend">' +
                '<span><span class="dot" style="background:#1890ff;"></span>\\u589e\\u503c\\u7a0e</span>' +
                '<span><span class="dot" style="background:#faad14;"></span>\\u4e2a\\u7a0e</span>' +
                '<span><span class="dot" style="background:#52c41a;"></span>\\u4f01\\u7a0e</span>' +
                '<span><span class="dot" style="background:#f5222d;"></span>\\u591a\\u7a0e\\u79cd\\u540c\\u65e5</span>' +
            '</div>' +
        '</div>';

    document.body.appendChild(modalOverlay);
}
